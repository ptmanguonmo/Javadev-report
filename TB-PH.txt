    private void btnThemTBActionPerformed(java.awt.event.ActionEvent evt) {                                          
        if(!tenTB.getText().equals("")){
            ThietBi t = new ThietBi();
            if(ThietBi.size() != 0){
                t.setID(((ThietBi)(ThietBi.get(ThietBi.size()-1))).getID()+1);
            t.setName(tenTB.getText());
            t.setXuatxu(XuatXu.getItemAt(XuatXu.getSelectedIndex()));
            t.setNam(Integer.parseInt(NamSX.getItemAt(NamSX.getSelectedIndex())));
            DefaultTableModel model = (DefaultTableModel) Table1.getModel();
            model.addRow(new Object[]{t.getID(), t.getName(), t.getXuatxu(), t.getNam()});
            ThietBi.add(t);
            ioFile.ghiFile(ThietBi, "DSTB.DAT");
            }
        }
        else JOptionPane.showMessageDialog(this, "Hãy nh?p d?y d? thông tin!");
      // TODO add your handling code here:
    }
